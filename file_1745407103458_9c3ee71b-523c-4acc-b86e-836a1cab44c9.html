    A([start]) --> B[documentproperties]
    B --> C[dataprocess]
    C --> D[documentproperties]
    D --> E[actionType+connectorType]
    E --> F{decision}
    F -- True --> G[message]
    F -- False --> H[documentproperties]
    G --> I[message]
    I --> J[map]
    J --> K[documentproperties]
    K --> L[actionType+connectorType]
    L --> M{branch}
    M --> N[documentproperties]
    M --> O[dataprocess]
    M --> P[actionType+connectorType]
    N --> Q[dataprocess]
    O --> R[documentproperties]
    P --> S[actionType+connectorType]
    Q --> T{branch}
    T --> U[dataprocess]
    T --> V[documentproperties]
    U --> W{decision}
    W -- True --> X[dataprocess]
    W -- False --> Y(stop)
    X --> Z[actionType+connectorType]
    Z --> AA[returndocuments]
    V --> AB[dataprocess]
    AB --> AC[dataprocess]
    AC --> Y
    R --> AD[dataprocess]
    AD --> AE[message]
    AE --> AF[dataprocess]
    AF --> AG[actionType+connectorType]
    AG --> AH[returndocuments]
    AG --> AI[message]
    AI --> AH
